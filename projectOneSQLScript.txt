USE jvillegas;

DELETE FROM ExerciseLifting WHERE ID=1;
DELETE FROM Person WHERE Age=0;
DELETE FROM Cardio WHERE ExerciseID=13;

CREATE TABLE Person (ID INT AUTO_INCREMENT PRIMARY KEY, FirstName VARCHAR(256), LastName VARCHAR(256), Age INT, StartWeight INT, Gender VARCHAR(25));

CREATE TABLE Food (Day VARCHAR(50), DayWeight INT, ID INT, FOREIGN KEY (ID) REFERENCES Person (ID) ON DELETE CASCADE, PRIMARY KEY (Day, ID));
CREATE TABLE Meal (FoodID INT PRIMARY KEY AUTO_INCREMENT, Time VARCHAR(25), Food VARCHAR(256), Calories INT, Day VARCHAR (50), 
ID INT, FOREIGN KEY (Day) REFERENCES Food (Day) ON DELETE CASCADE ON UPDATE CASCADE, FOREIGN KEY (ID) REFERENCES Person (ID) ON DELETE CASCADE);

CREATE TABLE Lifting (Day VARCHAR(50), ID INT, FOREIGN KEY (ID) REFERENCES Person (ID) ON DELETE CASCADE, PRIMARY KEY (Day, ID));
CREATE TABLE ExerciseLifting (ExerciseID INT AUTO_INCREMENT PRIMARY KEY, Day VARCHAR(50), Exercise VARCHAR(256), Weight INT, 
Reps INT, Sets INT, ID INT, FOREIGN KEY (ID) REFERENCES Person (ID), FOREIGN KEY (Day) REFERENCES Lifting (Day) ON DELETE CASCADE ON UPDATE CASCADE, 
UNIQUE (ExerciseID, Day));

CREATE TABLE Cardio (ExerciseID INT AUTO_INCREMENT PRIMARY KEY, Exercise VARCHAR(256), Time VARCHAR(25), Day VARCHAR(50), ID INT, FOREIGN KEY (ID) REFERENCES Person (ID) ON DELETE CASCADE);
